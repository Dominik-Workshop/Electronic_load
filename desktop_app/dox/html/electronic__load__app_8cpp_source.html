153uint32_t calculateCRC32(const uint8_t* data, size_t length) {
154 const uint32_t polynomial = 0xEDB88320;
155 uint32_t crc = 0xFFFFFFFF;
156
157 for (size_t i = 0; i < length; ++i) {
158 crc ^= data[i];
159 for (size_t j = 0; j < 8; ++j) {
160 crc = (crc >> 1) ^ (-(int)(crc & 1) & polynomial);
161 }
162 }
163 return crc ^ 0xFFFFFFFF;
164}

171 QStringList parts = QString(dataFromSerialPort).split(";");
172 if (parts.size() != 9) {
173 qDebug() << "Invalid data format. Expected 9 parts separated by semicolons.";
174 return;
175 }
176
177 if (parts[Command] == 'm') {
178 bool ok;
179 quint16 receivedVoltage = parts[MeasuredVoltage].toUShort(&ok);
180 quint16 receivedCurrent = parts[MeasuredCurrent].toUShort(&ok);
181 quint16 receivedTemperature = parts[MeasuredTemperature].toUShort(&ok);
182 quint16 receivedCutoffVoltage = parts[SetCutofffVoltage].toUShort(&ok);
183 quint16 receivedDischargeCurrent = parts[SetCurret].toUShort(&ok);
184 quint16 receivedIsLoadOn = parts[IsLoadOn].toUShort(&ok);
185 qint32 receivedTime = parts[Time].toLong(&ok);
186 uint32_t receivedCRC = parts[CRC].toUInt(&ok);
187
188 if (!ok) {
189 qDebug() << "Error converting received data";
190 return;
191 }
192
193 // Prepare the data for CRC calculation
194 uint8_t message[6 * sizeof(quint16) + sizeof(qint32)];
195 size_t index = 0;
196 memcpy(&message[index], &receivedVoltage, sizeof(quint16));
197 index += sizeof(quint16);
198 memcpy(&message[index], &receivedCurrent, sizeof(quint16));
199 index += sizeof(quint16);
200 memcpy(&message[index], &receivedTemperature, sizeof(quint16));
201 index += sizeof(quint16);
202 memcpy(&message[index], &receivedCutoffVoltage, sizeof(quint16));
203 index += sizeof(quint16);
204 memcpy(&message[index], &receivedDischargeCurrent, sizeof(quint16));
205 index += sizeof(quint16);
206 memcpy(&message[index], &receivedIsLoadOn, sizeof(quint16));
207 index += sizeof(quint16);
208 memcpy(&message[index], &receivedTime, sizeof(qint32));
209
210 uint32_t calculatedCRC = calculateCRC32(message, sizeof(message));
211
212 //qDebug() << "Received CRC (Decimal):" << receivedCRC;
213 //qDebug() << "Calculated CRC (Decimal):" << calculatedCRC;
214
215 if (receivedCRC == calculatedCRC) {
216 // qDebug() << "CRC Matched!";
217 } else {
218 qDebug() << "!!!!!!!CRC Mismatch!!!!!!!";
219 return;

221
222 // Update the set current and cutoff voltage if changed
223 setCurrent_A = parts[SetCurret].toFloat() / 1000;
224 setCutoffVoltage_V = parts[SetCutofffVoltage].toFloat() / 1000;
225 //qDebug() << setCurrent << " flag " << setCurrentEdited_flag;
