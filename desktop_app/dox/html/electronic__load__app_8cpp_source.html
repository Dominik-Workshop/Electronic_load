145uint32_t calculateCRC(const uint8_t* data, size_t length) {
146 const uint32_t polynomial = 0xEDB88320;
147 uint32_t crc = 0xFFFFFFFF;
148
149 for (size_t i = 0; i < length; ++i) {
150 crc ^= data[i];
151 for (size_t j = 0; j < 8; ++j) {
152 crc = (crc >> 1) ^ (-(int)(crc & 1) & polynomial);
153 }
154 }
155 return crc ^ 0xFFFFFFFF;
156}
