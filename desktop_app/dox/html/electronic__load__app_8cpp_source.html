168uint32_t calculateCRC32(const uint8_t* data, size_t length) {
169 const uint32_t polynomial = 0xEDB88320;
170 uint32_t crc = 0xFFFFFFFF;
171
172 for (size_t i = 0; i < length; ++i) {
173 crc ^= data[i];
174 for (size_t j = 0; j < 8; ++j) {
175 crc = (crc >> 1) ^ (-(int)(crc & 1) & polynomial);
176 }
177 }
178 return crc ^ 0xFFFFFFFF;
179}

185void Electronic_load_app::processReceivedData(){
186 QStringList parts = QString(dataFromSerialPort).split(";");
187 if (parts.size() != 9) {
188 qDebug() << "Invalid data format. Expected 9 parts separated by semicolons.";
189 return;
190 }
191
192 if (parts[Command] == 'm') {
193 bool ok;
194 quint16 receivedVoltage = parts[MeasuredVoltage].toUShort(&ok);
195 quint16 receivedCurrent = parts[MeasuredCurrent].toUShort(&ok);
196 quint16 receivedTemperature = parts[MeasuredTemperature].toUShort(&ok);
197 quint16 receivedCutoffVoltage = parts[SetCutofffVoltage].toUShort(&ok);
198 quint16 receivedDischargeCurrent = parts[SetCurret].toUShort(&ok);
199 quint16 receivedIsLoadOn = parts[IsLoadOn].toUShort(&ok);
200 qint32 receivedTime = parts[Time].toLong(&ok);
201 uint32_t receivedCRC = parts[CRC].toUInt(&ok);
202
203 if (!ok) {
204 qDebug() << "Error converting received data";
205 return;
206 }
207
208 // Prepare the data for CRC calculation
209 uint8_t message[6 * sizeof(quint16) + sizeof(qint32)];
210 size_t index = 0;
211 memcpy(&message[index], &receivedVoltage, sizeof(quint16));
212 index += sizeof(quint16);
213 memcpy(&message[index], &receivedCurrent, sizeof(quint16));
214 index += sizeof(quint16);
215 memcpy(&message[index], &receivedTemperature, sizeof(quint16));
216 index += sizeof(quint16);
217 memcpy(&message[index], &receivedCutoffVoltage, sizeof(quint16));
218 index += sizeof(quint16);
219 memcpy(&message[index], &receivedDischargeCurrent, sizeof(quint16));
220 index += sizeof(quint16);
221 memcpy(&message[index], &receivedIsLoadOn, sizeof(quint16));
222 index += sizeof(quint16);
223 memcpy(&message[index], &receivedTime, sizeof(qint32));
224
225 uint32_t calculatedCRC = calculateCRC32(message, sizeof(message));
226
227 //qDebug() << "Received CRC (Decimal):" << receivedCRC;
228 //qDebug() << "Calculated CRC (Decimal):" << calculatedCRC;
229
230 if (receivedCRC == calculatedCRC) {
231 // qDebug() << "CRC Matched!";
232 } else {
233 qDebug() << "!!!!!!!CRC Mismatch!!!!!!!";
234 return;
235 }
236
237 // Update the set current and cutoff voltage if changed
238 setCurrent_A = parts[SetCurret].toFloat() / 1000;
239 setCutoffVoltage_V = parts[SetCutofffVoltage].toFloat() / 1000;
240 //qDebug() << setCurrent << " flag " << setCurrentEdited_flag;
241 if(!setCurrentEdited_flag){
242 setCurrentEdited_flag = false;
243 ui->setCurrent->setText(QString::number(setCurrent_A, 'f', 3));
244 }
245
246 if (!setCutoffVoltageEdited_flag) {
247 ui->cutoffVoltage->setText(QString::number(setCutoffVoltage_V, 'f', 3));
248 }
249
250 // Add reading to measurements and update UI
251 float measuredVoltage = parts[MeasuredVoltage].toFloat() / 1000;
252 float measuredCurrent = parts[MeasuredCurrent].toFloat() / 1000;
253 int measuredTemperature = parts[MeasuredTemperature].toInt();
254 float elapsedTime = elapsedTimer.elapsed() / 1000.0;
255
256 measurements.addReading(measuredVoltage, measuredCurrent, elapsedTime, measuredTemperature);
257
258 ui->measuredVoltage->setText(QString::number(measuredVoltage, 'f', 3));
259 ui->measuredCurrent->setText(QString::number(measuredCurrent, 'f', 3));
260
261 isLoadOn = (parts[IsLoadOn] == "1");
262 if(isLoadOn){
263 ui->load_on_offfButton->setText(QObject::tr("Load ON"));
264 ui->load_on_offfButton->setStyleSheet("* { background-color: rgb(0,255,0) }");
265 }
266 else {
267 ui->load_on_offfButton->setStyleSheet("* { background-color: rgb(255,0,0) }");
268 ui->load_on_offfButton->setText(QObject::tr("Load OFF"));
269 }
270 }
271
272 if(measurements.nominalCapacity_mAh != 0) {
273 int capacityPercentage = 100 * measurements.capacity_mAh / measurements.nominalCapacity_mAh;
274 ui->batteryPercentageLabel->setText(QString::number(capacityPercentage) + "%");
275
276 // Calculate the color from red to green based on percentage, capping at 100 for the color calculation
277 int displayPercentage = std::min(capacityPercentage, 100);
278 ui->BatCapacityBar->setValue(displayPercentage);
279 int red = 255 * (100 - displayPercentage) / 100;
280 int green = 255 * displayPercentage / 100;
281 QString progressBarStyle = QString(
282 "QProgressBar {"
283 " background-color: rgb(136, 138, 133);"
284 " border: 1px solid grey;"
285 " border-radius: 5px;"
286 " text-align: center;"
287 "}"
288 "QProgressBar::chunk {"
289 " background-color: rgb(%1, %2, 0);"
290 " border-radius: 5px;"
291 "}"
292 ).arg(red).arg(green);
293
294 ui->BatCapacityBar->setStyleSheet(progressBarStyle);
295 ui->BatCapacityBar->update();
296 }
